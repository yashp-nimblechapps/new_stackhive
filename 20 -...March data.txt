Right now:
Your app controls permissions in UI.

Example:

But this is NOT secure.
Because anyone could:
Use Postman
Use Firebase REST API
Modify data directly
Bypass your UI

Security Rules = Backend Protection

They enforce:

✔ Who can read
✔ Who can write
✔ Who can delete
✔ Who can update

At Firestore server level.

Even if someone hacks your app — rules still protect data.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
  
		// =============================
    // USERS
    // =============================
    match /users/{userId} {

      // Any logged-in user can read profiles
      allow read: if request.auth != null;

      // User can create their own profile
      allow create: if request.auth != null
      	&& request.auth.uid == userId;

    	// User can update their own profile
      allow update: if request.auth != null
      	&& request.auth.uid == userId;

      // User can delete their own profile
    	allow delete: if request.auth != null
      	&& request.auth.uid == userId;
    }  	


    // QUESTIONS
      match /questions/{questionId} {

        allow read: if true;

        allow create: if request.auth != null
          && request.resource.data.userId == request.auth.uid;

        allow update: if request.auth != null
          && request.auth.uid == resource.data.userId;

        allow delete: if request.auth != null
          && request.auth.uid == resource.data.userId;
     
           

      // ANSWERS (Subcollection)
      match /answers/{answerId} {

        // Anyone can read answers
        allow read: if true;

        // Only authenticated users can create
        // CREATE ANSWER
        allow create: if request.auth != null
          && request.auth.uid == request.resource.data.userId
          && request.resource.data.voteCount == 0
          && request.resource.data.isBestAnswer == false
          && request.resource.data.keys().hasAll([
               'questionId',
               'userId',
               'content',
               'voteCount',
               'isBestAnswer',
               'createdAt'
             ]);

        // Only answer owner can update
        // UPDATE ANSWER (Only content allowed)
        allow update: if request.auth != null
          && (
               // 1 Answer owner editing content
               (
                 request.auth.uid == resource.data.userId
                 && request.resource.data.diff(resource.data)
                      .changedKeys()
                      .hasOnly(['content'])
               )

               ||

               // 2 Voting logic (voteCount only)
               (
                 request.resource.data.diff(resource.data)
                   .changedKeys()
                   .hasOnly(['voteCount'])
                 && request.resource.data.voteCount >= 0
								 && request.resource.data.voteCount <= 100000
               )

               ||

               // 3 Mark Best Answer (question owner only)
               (
                 request.resource.data.diff(resource.data)
                   .changedKeys()
                   .hasOnly(['isBestAnswer'])
                 && request.auth.uid ==
                    get(/databases/$(database)/documents/questions/$(questionId))
                      .data.userId
               )
             );

        // Delete allowed if:
        // - Answer owner
        // - OR Question owner
        allow delete: if request.auth != null
          && (
              request.auth.uid == resource.data.userId
              ||
              request.auth.uid ==
                get(/databases/$(database)/documents/questions/$(questionId))
                  .data.userId
            );

        // VOTES (Nested Subcollection)
        match /votes/{userId} {

          allow read: if true;

          // User can only write their own vote doc
          allow create, update, delete: if request.auth != null
            && request.auth.uid == userId;
        }
      }
    }
  }
}
